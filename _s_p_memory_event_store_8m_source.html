62- (nonnull NSArray<SPEmitterEvent *> *)emittableEventsWithQueryLimit:(NSUInteger)queryLimit {
63 @synchronized (self) {
64 NSUInteger setCount = [self.orderedSet count];
65 if (setCount <= 0) {
66 return @[];
67 }
68 NSUInteger len = MIN(queryLimit, setCount);
69 NSRange range = NSMakeRange(0, len);
70 SPEmitterEvent * __unsafe_unretained array[len];
71 [self.orderedSet getObjects:array range:range];
72 NSMutableArray<SPEmitterEvent *> *result = [[NSMutableArray alloc] initWithCapacity:len];
73 for (int i = 0; i < len; i++) {
74 [result addObject:array[i]];
75 }
76 return result;
77 }
