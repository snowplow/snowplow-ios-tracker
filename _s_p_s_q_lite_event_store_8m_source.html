57 + (NSArray<NSString *> *)removeUnsentEventsExceptForNamespaces:(NSArray<NSString *> *)allowedNamespaces {
58 #if SNOWPLOW_TARGET_TV
59  NSString *libraryPath = [NSSearchPathForDirectoriesInDomains(NSCachesDirectory, NSUserDomainMask, YES) objectAtIndex:0];
60 #else
61  NSString *libraryPath = [NSSearchPathForDirectoriesInDomains(NSLibraryDirectory, NSUserDomainMask, YES) objectAtIndex:0];
62 #endif
63  NSString *snowplowDirPath = [libraryPath stringByAppendingPathComponent:@"snowplow"];
64  NSArray<NSString *> *files = [NSFileManager.defaultManager contentsOfDirectoryAtPath:snowplowDirPath error:nil];
65  NSMutableArray<NSString *> *allowedFiles = [NSMutableArray new];
66  for (NSString *namespace in allowedNamespaces) {
67  NSRegularExpression *regex = [NSRegularExpression regularExpressionWithPattern:@"[^a-zA-Z0-9_]+" options:0 error:nil];
68  NSString *sqliteSuffix = [regex stringByReplacingMatchesInString:namespace options:0 range:NSMakeRange(0, namespace.length) withTemplate:@"-"];
69  NSString *sqliteFilename = [NSString stringWithFormat:@"snowplowEvents-%@.sqlite", sqliteSuffix];
70  [allowedFiles addObject:sqliteFilename];
71  }
72  NSMutableArray<NSString *> *removedFiles = [NSMutableArray new];
73  for (NSString *file in files) {
74  if (![allowedFiles containsObject:file]) {
75  NSString *pathToRemove = [snowplowDirPath stringByAppendingPathComponent:file];
76  [NSFileManager.defaultManager removeItemAtPath:pathToRemove error:nil];
77  [removedFiles addObject:file];
78  }
79  }
80  return removedFiles.copy;
81 }
