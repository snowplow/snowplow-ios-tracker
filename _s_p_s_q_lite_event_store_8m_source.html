56+ (NSArray<NSString *> *)removeUnsentEventsExceptForNamespaces:(NSArray<NSString *> *)allowedNamespaces {
57#if SNOWPLOW_TARGET_TV
58 NSString *libraryPath = [NSSearchPathForDirectoriesInDomains(NSCachesDirectory, NSUserDomainMask, YES) objectAtIndex:0];
59#else
60 NSString *libraryPath = [NSSearchPathForDirectoriesInDomains(NSLibraryDirectory, NSUserDomainMask, YES) objectAtIndex:0];
61#endif
62 NSString *snowplowDirPath = [libraryPath stringByAppendingPathComponent:@"snowplow"];
63 NSArray<NSString *> *files = [NSFileManager.defaultManager contentsOfDirectoryAtPath:snowplowDirPath error:nil];
64 NSMutableArray<NSString *> *allowedFiles = [NSMutableArray new];
65 for (NSString *namespace in allowedNamespaces) {
66 NSRegularExpression *regex = [NSRegularExpression regularExpressionWithPattern:@"[^a-zA-Z0-9_]+" options:0 error:nil];
67 NSString *sqliteSuffix = [regex stringByReplacingMatchesInString:namespace options:0 range:NSMakeRange(0, namespace.length) withTemplate:@"-"];
68 NSString *sqliteFilename = [NSString stringWithFormat:@"snowplowEvents-%@.sqlite", sqliteSuffix];
69 [allowedFiles addObject:sqliteFilename];
70 }
71 NSMutableArray<NSString *> *removedFiles = [NSMutableArray new];
72 for (NSString *file in files) {
73 if (![allowedFiles containsObject:file]) {
74 NSString *pathToRemove = [snowplowDirPath stringByAppendingPathComponent:file];
75 [NSFileManager.defaultManager removeItemAtPath:pathToRemove error:nil];
76 [removedFiles addObject:file];
77 }
78 }
79 return removedFiles.copy;
80}

86- (instancetype)initWithNamespace:(NSString *)namespace limit:(NSUInteger)limit {
87 if (self = [super init]) {
88 self.namespace = namespace;
89 self.sendLimit = limit;
90
91#if SNOWPLOW_TARGET_TV
92 NSString *libraryPath = [NSSearchPathForDirectoriesInDomains(NSCachesDirectory, NSUserDomainMask, YES) objectAtIndex:0];
93#else
94 NSString *libraryPath = [NSSearchPathForDirectoriesInDomains(NSLibraryDirectory, NSUserDomainMask, YES) objectAtIndex:0];
95#endif
96 // Create snowplow subdirectory if it doesn't exist
97 NSString *snowplowDirPath = [libraryPath stringByAppendingPathComponent:@"snowplow"];
98 [[NSFileManager defaultManager] createDirectoryAtPath:snowplowDirPath withIntermediateDirectories:YES attributes:nil error:nil];
99
100 // Create path for the database
101 NSRegularExpression *regex = [NSRegularExpression regularExpressionWithPattern:@"[^a-zA-Z0-9_]+" options:0 error:nil];
102 NSString *sqliteSuffix = [regex stringByReplacingMatchesInString:namespace options:0 range:NSMakeRange(0, namespace.length) withTemplate:@"-"];
103 self.sqliteFilename = [NSString stringWithFormat:@"snowplowEvents-%@.sqlite", sqliteSuffix];
104 self.dbPath = [snowplowDirPath stringByAppendingPathComponent:self.sqliteFilename];
105
106 // Migrate old database if it exists
107 NSString *oldDbPath = [libraryPath stringByAppendingPathComponent:@"snowplowEvents.sqlite"];
108 if ([[NSFileManager defaultManager] fileExistsAtPath:oldDbPath]) {
109 [[NSFileManager defaultManager] moveItemAtPath:oldDbPath toPath:self.dbPath error:nil];
110 }
111
112 // Create database
113 self.queue = [FMDatabaseQueue databaseQueueWithPath:self.dbPath];
114 [self createTable];
115 }
116 return self;
117}

216- (SPEmitterEvent *) getEventWithId:(long long int)id_ {
217 __block SPEmitterEvent *event = nil;
218 [self.queue inDatabase:^(FMDatabase *db) {
219 if ([db open]) {
220 FMResultSet *s = [db executeQuery:_querySelectId, [NSNumber numberWithLongLong:id_]];
221 while ([s next]) {
222 NSData *data = [s dataForColumn:@"eventData"];
223 NSDictionary *dict = [SPJSONSerialization deserializeData:data];
224 if (!dict) {
225 continue;
226 }
227 SPPayload *payload = [[SPPayload alloc] initWithNSDictionary:dict];
228 event = [[SPEmitterEvent alloc] initWithPayload:payload storeId:id_];
229 }
230 [s close];
231 }
232 }];
233 return event;
234}

245- (NSArray<SPEmitterEvent *> *) getAllEventsWithQuery:(NSString *)query {
246 __block NSMutableArray *res = [[NSMutableArray alloc] init];
247 [self.queue inDatabase:^(FMDatabase *db) {
248 if ([db open]) {
249 FMResultSet *s = [db executeQuery:query];
250 while ([s next]) {
251 long long int index = [s longLongIntForColumn:@"ID"];
252 NSData *data = [s dataForColumn:@"eventData"];
253 NSDictionary *dict = [SPJSONSerialization deserializeData:data];
254 if (!dict) {
255 continue;
256 }
257 SPPayload *payload = [[SPPayload alloc] initWithNSDictionary:dict];
258 SPEmitterEvent *event = [[SPEmitterEvent alloc] initWithPayload:payload storeId:index];
259 [res addObject:event];
260 }
261 [s close];
262 }
263 }];
264 return res;
265}
