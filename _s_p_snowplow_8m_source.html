65+ (id<SPTrackerController>)createTrackerWithNamespace:(NSString *)namespace network:(SPNetworkConfiguration *)networkConfiguration configurations:(NSArray<SPConfiguration *> *)configurations {
66 SPServiceProvider *serviceProvider = [[SPSnowplow sharedInstance].serviceProviderInstances objectForKey:namespace];
67 if (serviceProvider) {
68 [serviceProvider resetWithConfigurations:[configurations arrayByAddingObject:networkConfiguration]];
69 } else {
70 serviceProvider = [[SPServiceProvider alloc] initWithNamespace:namespace network:networkConfiguration configurations:configurations];
71 [SPSnowplow.sharedInstance registerInstance:serviceProvider];
72 }
73 return serviceProvider.trackerController;

132 @synchronized (self) {
133 NSString *namespace = serviceProvider.namespace;
134 BOOL isOverriding = [self.serviceProviderInstances objectForKey:namespace] != nil;
135 [self.serviceProviderInstances setObject:serviceProvider forKey:namespace];

145+ (NSArray<NSString *> *)createTrackersWithConfigurationBundles:(NSArray<SPConfigurationBundle *> *)bundles {
146 NSMutableArray<NSString *> *namespaces = [NSMutableArray new];
147 for (SPConfigurationBundle *bundle in bundles) {
148 @synchronized (SPSnowplow.class) {
149 if (!bundle.networkConfiguration) {
150 // remove tracker if it exists
151 id<SPTrackerController> tracker = [SPSnowplow trackerByNamespace:bundle.namespace];
152 if (tracker) {
153 [SPSnowplow removeTracker:tracker];
154 }
155 } else {
156 [SPSnowplow createTrackerWithNamespace:bundle.namespace
157 network:bundle.networkConfiguration
158 configurations:bundle.configurations
159 ];
160 [namespaces addObject:bundle.namespace];
161 }
162 }
163 }
164 return namespaces;
165}
