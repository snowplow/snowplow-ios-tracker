214 - (NSDictionary *) payload {
215  NSMutableDictionary * event = [[NSMutableDictionary alloc] init];
216  [event setObject:_title forKey:kSPPnTitle];
217  [event setObject:_body forKey:kSPPnBody];
218  [event setValue:_badge forKey:kSPPnBadge];
219  if (_subtitle != nil) {
220  [event setObject:_subtitle forKey:kSPPnSubtitle];
221  }
222  if (_subtitle != nil) {
223  [event setObject:_subtitle forKey:kSPPnSubtitle];
224  }
225  if (_sound != nil) {
226  [event setObject:_sound forKey:kSPPnSound];
227  }
228  if (_launchImageName != nil) {
229  [event setObject:_launchImageName forKey:kSPPnLaunchImageName];
230  }
231  if (_userInfo != nil) {
232  NSMutableDictionary * aps = nil;
233  NSMutableDictionary * newUserInfo = nil;
234 
235  // modify contentAvailable value "1" and "0" to @YES and @NO to comply with schema
236  if (![[_userInfo valueForKeyPath:@"aps.contentAvailable"] isEqual:nil] &&
237  [[_userInfo objectForKey:@"aps"] isKindOfClass:[NSDictionary class]]) {
238  aps = [[NSMutableDictionary alloc] initWithDictionary:_userInfo[@"aps"]];
239 
240  if ([[_userInfo valueForKeyPath:@"aps.contentAvailable"] isEqual:@1]) {
241  [aps setObject:@YES forKey:@"contentAvailable"];
242  } else if ([[_userInfo valueForKeyPath:@"aps.contentAvailable"] isEqual:@0]) {
243  [aps setObject:@NO forKey:@"contentAvailable"];
244  }
245  newUserInfo = [[NSMutableDictionary alloc] initWithDictionary:_userInfo];
246  [newUserInfo setObject:aps forKey:@"aps"];
247  }
248  [event setObject:[[NSDictionary alloc] initWithDictionary:newUserInfo] forKey:kSPPnUserInfo];
249  }
250  if (_attachments != nil) {
251  [event setObject:_attachments forKey:kSPPnAttachments];
252  }
253 
254  return [[NSDictionary alloc] initWithDictionary:event copyItems:YES];
255 }
