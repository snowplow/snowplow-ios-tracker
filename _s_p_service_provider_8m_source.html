91- (instancetype)initWithNamespace:(NSString *)namespace network:(SPNetworkConfiguration *)networkConfiguration configurations:(NSArray<SPConfiguration *> *)configurations {
92 if (self = [super init]) {
93 [self initializeConfigurationUpdates];
94 self.namespace = namespace;
95 self.networkConfigurationUpdate.sourceConfig = networkConfiguration;
96 [self processConfigurations:configurations];
97 if (!self.trackerConfigurationUpdate.sourceConfig) {
98 self.trackerConfigurationUpdate.sourceConfig = [SPTrackerConfiguration new];
99 }
100 [self tracker]; // Build tracker to initialize NotificationCenter receivers
101 }
102 return self;
103}

123- (void)processConfigurations:(NSArray<SPConfiguration *> *)configurations {
124 for (SPConfiguration *configuration in configurations) {
125 if ([configuration isKindOfClass:SPNetworkConfiguration.class]) {
126 self.networkConfigurationUpdate.sourceConfig = (SPNetworkConfiguration *)configuration;
127 continue;
128 }
129 if ([configuration isKindOfClass:SPTrackerConfiguration.class]) {
130 self.trackerConfigurationUpdate.sourceConfig = (SPTrackerConfiguration *)configuration;
131 continue;
132 }
133 if ([configuration isKindOfClass:SPSubjectConfiguration.class]) {
134 self.subjectConfigurationUpdate.sourceConfig = (SPSubjectConfiguration *)configuration;
135 continue;
136 }
137 if ([configuration isKindOfClass:SPSessionConfiguration.class]) {
138 self.sessionConfigurationUpdate.sourceConfig = (SPSessionConfiguration *)configuration;
139 continue;
140 }
141 if ([configuration isKindOfClass:SPEmitterConfiguration.class]) {
142 self.emitterConfigurationUpdate.sourceConfig = (SPEmitterConfiguration *)configuration;
143 continue;
144 }
145 if ([configuration isKindOfClass:SPGDPRConfiguration.class]) {
146 self.gdprConfigurationUpdate.sourceConfig = (SPGDPRConfiguration *)configuration;
147 continue;
148 }
149 if ([configuration isKindOfClass:SPGlobalContextsConfiguration.class]) {
150 self.globalContextConfiguration = (SPGlobalContextsConfiguration *)configuration;
151 continue;
152 }
153 }

268- (SPEmitter *)makeEmitter {
269 SPNetworkConfigurationUpdate *networkConfig = self.networkConfigurationUpdate;
270 SPEmitterConfigurationUpdate *emitterConfig = self.emitterConfigurationUpdate;
271 SPEmitter *emitter = [SPEmitter build:^(id<SPEmitterBuilder> builder) {
272 if (networkConfig.networkConnection) {
273 [builder setNetworkConnection:networkConfig.networkConnection];
274 } else {
275 [builder setHttpMethod:networkConfig.method];
276 [builder setProtocol:networkConfig.protocol];
277 [builder setUrlEndpoint:networkConfig.endpoint];
278 }
279 [builder setCustomPostPath:networkConfig.customPostPath];
280 [builder setRequestHeaders:networkConfig.requestHeaders];
281 if (emitterConfig) {
282 [builder setEmitRange:emitterConfig.emitRange];
283 [builder setBufferOption:emitterConfig.bufferOption];
284 [builder setEventStore:emitterConfig.eventStore];
285 [builder setByteLimitPost:emitterConfig.byteLimitPost];
286 [builder setByteLimitGet:emitterConfig.byteLimitGet];
287 [builder setEmitThreadPoolSize:emitterConfig.threadPoolSize];
288 [builder setCallback:emitterConfig.requestCallback];
289 }
290 }];
291 if (emitterConfig && emitterConfig.isPaused) {
292 [emitter pauseEmit];
293 }
294 return emitter;
295}

297- (SPTracker *)makeTracker {
298 SPEmitter *emitter = self.emitter;
299 SPSubject *subject = self.subject;
300 SPTrackerConfiguration *trackerConfig = self.trackerConfigurationUpdate;
301 SPSessionConfiguration *sessionConfig = self.sessionConfigurationUpdate;
302 SPGlobalContextsConfiguration *gcConfig = self.globalContextConfiguration;
303 SPTracker *tracker = [SPTracker build:^(id<SPTrackerBuilder> builder) {
304 [builder setTrackerNamespace:self.namespace];
305 [builder setEmitter:emitter];
306 [builder setSubject:subject];
307 [builder setAppId:trackerConfig.appId];
308 [builder setTrackerVersionSuffix:trackerConfig.trackerVersionSuffix];
309 [builder setBase64Encoded:trackerConfig.base64Encoding];
310 [builder setLogLevel:trackerConfig.logLevel];
311 [builder setLoggerDelegate:trackerConfig.loggerDelegate];
312 [builder setDevicePlatform:trackerConfig.devicePlatform];
313 [builder setSessionContext:trackerConfig.sessionContext];
314 [builder setApplicationContext:trackerConfig.applicationContext];
315 [builder setDeepLinkContext:trackerConfig.deepLinkContext];
316 [builder setScreenContext:trackerConfig.screenContext];
317 [builder setAutotrackScreenViews:trackerConfig.screenViewAutotracking];
318 [builder setLifecycleEvents:trackerConfig.lifecycleAutotracking];
319 [builder setInstallEvent:trackerConfig.installAutotracking];
320 [builder setExceptionEvents:trackerConfig.exceptionAutotracking];
321 [builder setTrackerDiagnostic:trackerConfig.diagnosticAutotracking];
322 if (sessionConfig) {
323 [builder setBackgroundTimeout:sessionConfig.backgroundTimeoutInSeconds];
324 [builder setForegroundTimeout:sessionConfig.foregroundTimeoutInSeconds];
325 }
326 if (gcConfig) {
327 [builder setGlobalContextGenerators:gcConfig.contextGenerators];
328 }
329 SPGDPRConfigurationUpdate *gdprConfig = self.gdprConfigurationUpdate;
330 if (gdprConfig.sourceConfig) {
331 [builder setGdprContextWithBasis:gdprConfig.basisForProcessing documentId:gdprConfig.documentId documentVersion:gdprConfig.documentVersion documentDescription:gdprConfig.documentDescription];
332 }
333 }];
334 if (self.trackerConfigurationUpdate.isPaused) {
335 [tracker pauseEventTracking];
336 }
337 if (tracker.session) {
338 if (self.sessionConfigurationUpdate.isPaused) {
339 [tracker.session stopChecker];
340 }
341 if (self.sessionConfigurationUpdate.onSessionStateUpdate) {
342 tracker.session.onSessionStateUpdate = self.sessionConfigurationUpdate.onSessionStateUpdate;
343 }
344 }
345 return tracker;
346}
