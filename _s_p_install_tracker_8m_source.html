28- (id)init {
29 if (self = [super init]) {
30 NSUserDefaults * userDefaults = [NSUserDefaults standardUserDefaults];
31 if ([userDefaults objectForKey:kSPInstalledBefore] == nil) {
32 // mark the install if there's no value in userDefaults
33 [userDefaults setObject:@YES forKey:kSPInstalledBefore];
34 [userDefaults setObject:[NSDate new] forKey:kSPInstallTimestamp];
35 // since the value was missing in userDefaults, we're assuming this is a new install
36 self.isNewInstall = YES;
37 } else {
38 // if there's an object in standardUserDefaults - someone has been there!
39 self.isNewInstall = NO;
40 }
41 return self;
42 }
43 return nil;
44}

46- (NSDate *)previousInstallTimestamp {
47 NSUserDefaults *userDefaults = [NSUserDefaults standardUserDefaults];
48 id value = [userDefaults objectForKey:kSPInstallTimestamp];
49 if (!value) {
50 return nil;
51 } else if ([value isKindOfClass:NSDate.class]) { // v2.0 format
52 return (NSDate *)value;
53 } else if ([value isKindOfClass:NSNumber.class]) { // v1.7 format
54 NSTimeInterval timeInterval = ((NSNumber *)value).doubleValue / 1000;
55 return [[NSDate alloc] initWithTimeIntervalSince1970:timeInterval];
56 }
57 return nil;
58}

60- (void)clearPreviousInstallTimestamp {
61 NSUserDefaults * userDefaults = [NSUserDefaults standardUserDefaults];
62 [userDefaults removeObjectForKey:kSPInstallTimestamp];
63}

65- (void)saveBuildAndVersion {
66 NSString * build = [SPUtilities getAppBuild];
67 NSString * version = [SPUtilities getAppVersion];
68 if (build && version) {
69 NSUserDefaults * userDefaults = [NSUserDefaults standardUserDefaults];
70 [userDefaults setObject:build forKey:kSPPreviousInstallBuild];
71 [userDefaults setObject:version forKey:kSPPreviousInstallVersion];
72 }
