42- (id)initWithRule:(NSString *)rule {

58- (BOOL)matchWithUri:(NSString *)uri {
59 if (!uri) {
60 return NO;
61 }
62 NSArray<NSString *> *uriParts = [self partsFromUri:uri regexPattern:kUriPattern];
63 if (uriParts.count < _ruleParts.count) {
64 return NO;
65 }
66 // Check vendor part
67 NSArray<NSString *> *ruleVendor = [_ruleParts[0] componentsSeparatedByString:@"."];
68 NSArray<NSString *> *uriVendor = [uriParts[0] componentsSeparatedByString:@"."];
69 if (uriVendor.count != ruleVendor.count) {
70 return NO;
71 }
72 NSUInteger index = 0;
73 for (NSString *ruleVendorPart in ruleVendor) {
74 if (![@"*" isEqualToString:ruleVendorPart] && ![uriVendor[index] isEqualToString:ruleVendorPart]) {
75 return NO;
76 }
77 index++;
78 }
79 // Check the rest of the rule
80 index = 1;
81 for (NSString *rulePart in [_ruleParts subarrayWithRange:NSMakeRange(1, _ruleParts.count-1)]) {
82 if (![@"*" isEqualToString:rulePart] && ![uriParts[index] isEqualToString:rulePart]) {
83 return NO;
84 }
85 index++;
86 }
87 return YES;
88}

92- (nullable NSArray<NSString *> *)partsFromUri:(NSString *)uri regexPattern:(NSString *)pattern {
93 NSError *error = NULL;
94 NSRegularExpression *regex = [NSRegularExpression regularExpressionWithPattern:pattern options:0 error:&error];
95 NSTextCheckingResult *match = [regex firstMatchInString:uri options:0 range:NSMakeRange(0, uri.length)];
96 NSMutableArray *parts = [NSMutableArray arrayWithCapacity:6];
97 if (!match) {
98 return nil;
99 }
100 for (int i = 1; i < match.numberOfRanges; i++) {
101 if (i > 6) {
102 return nil;
103 }
104 NSString *part = [uri substringWithRange:[match rangeAtIndex:i]];
105 [parts setObject:part.copy atIndexedSubscript:i-1];
106 }
107 return parts;
108}

110- (BOOL)validateVendor:(NSString *)vendor {
111 // the components array will be generated like this from vendor:
112 // "com.acme.marketing" => ["com", "acme", "marketing"]
113 NSArray<NSString *> *components = [vendor componentsSeparatedByString:@"."];
114 // check that vendor doesn't begin or end with period
115 // e.g. ".snowplowanalytics.snowplow." => ["", "snowplowanalytics", "snowplow", ""]
116 if (components.count > 1 && (!components[0].length || !components[components.count-1].length)) {
117 return NO;
118 }
119 // reject vendors with criteria that are too broad & don't make sense, i.e. "*.*.marketing"
120 if ([@"*" isEqualToString:components[0]] || [@"*" isEqualToString:components[1]]) {
121 return NO;
122 }
123 // now validate the remaining parts, vendors should follow matching that never breaks trailing specificity
124 // in other words, once we use an asterisk, we must continue using asterisks for parts or stop
125 // e.g. "com.acme.marketing.*.*" is allowed, but "com.acme.*.marketing.*" or "com.acme.*.marketing" is forbidden
126 if (components.count <= 2) return YES;
127 // trailingComponents are the remaining parts after the first two
128 NSArray<NSString *> *trailingComponents = [components subarrayWithRange:NSMakeRange(2, components.count-2)];
129 BOOL asterisk = NO;
130 for (NSString *part in trailingComponents) {
131 if ([@"*" isEqualToString:part]) { // mark when we've found a wildcard
132 asterisk = true;
133 } else if (asterisk) { // invalid when alpha parts come after wildcard
134 return NO;
135 }
136 }
137 return YES;
138}
