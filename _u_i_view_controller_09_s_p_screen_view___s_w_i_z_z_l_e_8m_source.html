68 - (void) SP_viewDidAppear:(BOOL)animated {

90 - (NSString *) _SP_getSnowplowId {
91  NSString * propertyName = @"snowplowId";
92  SEL selector = NSSelectorFromString(propertyName);
93  BOOL propertyExists = [self respondsToSelector:selector];
94  if (propertyExists) {
95  id value = [self valueForKey:propertyName];
96  if (value && ([value isKindOfClass:[NSString class]] || [NSStringFromClass([value class]) isEqualToString:@"Swift.String"])) {
97  return value;

132 - (SPScreenType) _SP_getViewControllerType:(UIViewController *)viewController {
133  if ([viewController isKindOfClass:[UINavigationController class]]) {
134  return SPScreenTypeNavigation;

149  if (uiPopoverPresentationControllerClass && [viewController isKindOfClass:uiPopoverPresentationControllerClass]) {
150  return SPScreenTypePopoverPresentation;

159 - (UIViewController *) _SP_topViewController {
160  UIWindow *keyWindow = nil;
161  NSArray<UIWindow *> *windows = [UIApplication sharedApplication].windows;
162  for (UIWindow *window in windows) {
163  if (window.isKeyWindow) {
164  keyWindow = window;
165  break;
166  }
167  }
168  if (!keyWindow) {
169  return nil;
170  }
171  return [self _SP_topViewController:keyWindow.rootViewController];
172 }

174 - (UIViewController *) _SP_topViewController:(UIViewController *)rootViewController {
175  if ([rootViewController isKindOfClass:[UINavigationController class]]) {
176  UINavigationController *navigationController = (UINavigationController *)rootViewController;
177  return [self _SP_topViewController:[navigationController.viewControllers lastObject]];
178  }
179 
180  if ([rootViewController isKindOfClass:[UITabBarController class]]) {
181  UITabBarController *tabController = (UITabBarController *)rootViewController;
182  return [self _SP_topViewController:tabController.selectedViewController];
183  }
184 
185  UIViewController *presentedViewController = rootViewController.presentedViewController;
186  if (presentedViewController != nil) {
187  return [self _SP_topViewController:presentedViewController];
188  }
189
