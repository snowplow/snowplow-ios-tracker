36+ (void) load {
37 static dispatch_once_t onceToken;
38 dispatch_once(&onceToken, ^{
39 //* Then swizzle */
40 Class class = [self class];
41
42 SEL originalSelector = @selector(viewDidAppear:);
43 SEL swizzledSelector = @selector(SP_viewDidAppear:);
44
45 Method originalMethod = class_getInstanceMethod(class, originalSelector);
46 Method swizzledMethod = class_getInstanceMethod(class, swizzledSelector);
47
48 BOOL didAddMethod =
49 class_addMethod(class,
50 originalSelector,
51 method_getImplementation(swizzledMethod),
52 method_getTypeEncoding(swizzledMethod));
53
54 if (didAddMethod) {
55 class_replaceMethod(class,
56 swizzledSelector,
57 method_getImplementation(originalMethod),
58 method_getTypeEncoding(originalMethod));
59 } else {
60 method_exchangeImplementations(originalMethod, swizzledMethod);
61 }
62 });
63}

67- (void) SP_viewDidAppear:(BOOL)animated {
68 [self SP_viewDidAppear:animated];
69
70 // Construct userInfo
71 NSMutableDictionary * userInfo = [[NSMutableDictionary alloc] init];
72 userInfo[@"viewControllerClassName"] = NSStringFromClass([self class]);
73 userInfo[@"topViewControllerClassName"] = NSStringFromClass([[self _SP_topViewController] class]);
74 // `name` is set to snowplowId class instance variable if it exists (hence no @"id" in userInfo)
75 userInfo[@"name"] = [self _SP_getViewControllerName];
76 userInfo[@"type"] = [[NSNumber alloc] initWithInteger:[self _SP_getTopViewControllerType]];
77
78 // Send notification to tracker
79 [[NSNotificationCenter defaultCenter]
80 postNotificationName:@"SPScreenViewDidAppear"
81 object:self
82 userInfo:userInfo];
83}

89- (NSString *) _SP_getSnowplowId {
90 NSString * propertyName = @"snowplowId";
91 SEL selector = NSSelectorFromString(propertyName);
92 BOOL propertyExists = [self respondsToSelector:selector];
93 if (propertyExists) {
94 id value = [self valueForKey:propertyName];
95 if (value && ([value isKindOfClass:[NSString class]] || [NSStringFromClass([value class]) isEqualToString:@"Swift.String"])) {
96 return value;
97 }
98 }
99 return nil;
100}

102- (NSString *) _SP_getViewControllerName {
103 NSString * viewControllerName = [self _SP_getViewControllerName:self];
104 NSString * topViewControllerName = [self _SP_getViewControllerName:[self _SP_topViewController]];
105
106 if ([self _SP_validateString:viewControllerName]) {
107 return viewControllerName;
108 } else if ([self _SP_validateString:topViewControllerName]) {
109 return topViewControllerName;
110 }
111
112 return @"Unknown";
113}

115- (NSString *) _SP_getViewControllerName:(UIViewController *)viewController {
116 // check if there's an instance variable snowplowId
117 NSString * viewControllerSnowplowId = [viewController _SP_getSnowplowId];

119 return viewControllerSnowplowId;

131- (SPScreenType) _SP_getViewControllerType:(UIViewController *)viewController {
132 if ([viewController isKindOfClass:[UINavigationController class]]) {
133 return SPScreenTypeNavigation;
134 }
135 if ([viewController isKindOfClass:[UITabBarController class]]) {
136 return SPScreenTypeTabBar;
137 }
138 if (viewController.presentedViewController) {
139 return SPScreenTypeModal;
140 }
141 if ([viewController isKindOfClass:[UIPageViewController class]]) {
142 return SPScreenTypePageView;
143 }
144 if ([viewController isKindOfClass:[UISplitViewController class]]) {
145 return SPScreenTypeSplitView;
146 }
147 Class uiPopoverPresentationControllerClass = NSClassFromString(@"UIPopoverPresentationController");
148 if (uiPopoverPresentationControllerClass && [viewController isKindOfClass:uiPopoverPresentationControllerClass]) {
149 return SPScreenTypePopoverPresentation;
150 }
151 return SPScreenTypeDefault;
152}

158- (UIViewController *) _SP_topViewController {
159 UIWindow *keyWindow = nil;
160 NSArray<UIWindow *> *windows = [UIApplication sharedApplication].windows;
161 for (UIWindow *window in windows) {
162 if (window.isKeyWindow) {
163 keyWindow = window;
164 break;
165 }
166 }
167 if (!keyWindow) {
168 return nil;
169 }
170 return [self _SP_topViewController:keyWindow.rootViewController];
171}
172
173- (UIViewController *) _SP_topViewController:(UIViewController *)rootViewController {
174 if ([rootViewController isKindOfClass:[UINavigationController class]]) {
175 UINavigationController *navigationController = (UINavigationController *)rootViewController;
176 return [self _SP_topViewController:[navigationController.viewControllers lastObject]];
177 }
178
179 if ([rootViewController isKindOfClass:[UITabBarController class]]) {
180 UITabBarController *tabController = (UITabBarController *)rootViewController;
181 return [self _SP_topViewController:tabController.selectedViewController];
182 }
183
184 UIViewController *presentedViewController = rootViewController.presentedViewController;
185 if (presentedViewController != nil) {
186 return [self _SP_topViewController:presentedViewController];
187 }
188
189 return rootViewController;
190}
