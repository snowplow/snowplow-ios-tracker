67 - (void) setup {
68  // Decode url to extract protocol
69  NSURL *url = [[NSURL alloc] initWithString:_urlString];
70  NSString *endpoint = _urlString;
71  if ([url.scheme isEqualToString:@"https"]) {
72  _protocol = SPProtocolHttps;
73  } else if ([url.scheme isEqualToString:@"http"]) {
74  _protocol = SPProtocolHttp;
75  } else {
76  _protocol = SPProtocolHttps;
77  endpoint = [NSString stringWithFormat:@"https://%@", _urlString];
78  }
79 
80  // Configure
81  NSString *urlPrefix = _protocol == SPProtocolHttp ? @"http://" : @"https://";
82  NSString *urlSuffix = _httpMethod == SPHttpMethodGet ? kSPEndpointGet : kSPEndpointPost;
83  if (_customPostPath && _httpMethod == SPHttpMethodPost) {
84  urlSuffix = _customPostPath;
85  }
86  _urlEndpoint = [NSURL URLWithString:[NSString stringWithFormat:@"%@%@", endpoint, urlSuffix]];
87 
88  // Log
89  if ([_urlEndpoint scheme] && [_urlEndpoint host]) {
90  SPLogDebug(@"Emitter URL created successfully '%@'", _urlEndpoint);
91  } else {
92  SPLogDebug(@"Invalid emitter URL: '%@'", _urlEndpoint);
93  }
94  NSUserDefaults *userDefaults = [NSUserDefaults standardUserDefaults];
95  [userDefaults setObject:endpoint forKey:kSPErrorTrackerUrl];
96  [userDefaults setObject:urlSuffix forKey:kSPErrorTrackerProtocol];
97  [userDefaults setObject:urlPrefix forKey:kSPErrorTrackerMethod];
98 
99  _builderFinished = YES;
100 }

151 - (NSArray<SPRequestResult *> *)sendRequests:(NSArray<SPRequest *> *)requests {
152  NSMutableArray<SPRequestResult *> *results = [NSMutableArray new];
153 
154  for (SPRequest *request in requests) {
155  NSMutableURLRequest *urlRequest = _httpMethod == SPHttpMethodGet
156  ? [self buildGetRequest:request]
157  : [self buildPostRequest:request];
158 
159  [_dataOperationQueue addOperationWithBlock:^{
160  //source: https://forums.developer.apple.com/thread/11519
161  __block NSHTTPURLResponse *httpResponse = nil;
162  __block NSError *connectionError = nil;
163  dispatch_semaphore_t sem;
164 
165  sem = dispatch_semaphore_create(0);
166 
167  [[[NSURLSession sharedSession] dataTaskWithRequest:urlRequest
168  completionHandler:^(NSData *data, NSURLResponse *urlResponse, NSError *error) {
169 
170  connectionError = error;
171  httpResponse = (NSHTTPURLResponse*)urlResponse;
172  dispatch_semaphore_signal(sem);
173  }] resume];
174 
175  dispatch_semaphore_wait(sem, DISPATCH_TIME_FOREVER);
176 
177  BOOL isSuccessful = [httpResponse statusCode] >= 200 && [httpResponse statusCode] < 300;
178  if (!isSuccessful) {
179  SPLogError(@"Connection error: %@", connectionError);
180  }
181  SPRequestResult *result = [[SPRequestResult alloc] initWithSuccess:isSuccessful storeIds:request.emitterEventIds];
182 
183  @synchronized (results) {
184  [results addObject:result];
185  }
186  }];
187  }
188  [_dataOperationQueue waitUntilAllOperationsAreFinished];
189  return results;
190 }

194 - (NSMutableURLRequest *)buildPostRequest:(SPRequest *)request {
195  NSData *requestData = [NSJSONSerialization dataWithJSONObject:[request.payload getAsDictionary] options:0 error:nil];
196  NSMutableURLRequest *urlRequest = [NSMutableURLRequest requestWithURL:[NSURL URLWithString:_urlEndpoint.absoluteString]];
197  [urlRequest setValue:[NSString stringWithFormat:@"%@", @(requestData.length).stringValue] forHTTPHeaderField:@"Content-Length"];
198  [urlRequest setValue:kSPAcceptContentHeader forHTTPHeaderField:@"Accept"];
199  [urlRequest setValue:kSPContentTypeHeader forHTTPHeaderField:@"Content-Type"];
200  [self applyValuesAndHeaderFields:_requestHeaders toRequest:urlRequest];
201  [urlRequest setHTTPMethod:@"POST"];
202  [urlRequest setHTTPBody:requestData];
203  return urlRequest;
204 }
