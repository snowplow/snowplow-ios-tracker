68- (void) setup {
69 // Decode url to extract protocol
70 NSURL *url = [[NSURL alloc] initWithString:_urlString];
71 NSString *endpoint = _urlString;
72 if ([url.scheme isEqualToString:@"https"]) {
73 _protocol = SPProtocolHttps;
74 } else if ([url.scheme isEqualToString:@"http"]) {
75 _protocol = SPProtocolHttp;
76 } else {
77 _protocol = SPProtocolHttps;
78 endpoint = [NSString stringWithFormat:@"https://%@", _urlString];
79 }
80
81 // Configure
82 NSString *urlPrefix = _protocol == SPProtocolHttp ? @"http://" : @"https://";
83 NSString *urlSuffix = _httpMethod == SPHttpMethodGet ? kSPEndpointGet : kSPEndpointPost;
84 if (_customPostPath && _httpMethod == SPHttpMethodPost) {
85 urlSuffix = _customPostPath;
86 }
87
88 // Remove trailing slashes from endpoint to avoid double slashes when appending path
89 endpoint = [endpoint stringByTrimmingCharactersInSet:[NSCharacterSet characterSetWithCharactersInString:@"/"]];
90
91 _urlEndpoint = [[NSURL URLWithString:endpoint] URLByAppendingPathComponent:urlSuffix];
92
93 // Log
94 if ([_urlEndpoint scheme] && [_urlEndpoint host]) {
95 SPLogDebug(@"Emitter URL created successfully '%@'", _urlEndpoint);
96 } else {
97 SPLogDebug(@"Invalid emitter URL: '%@'", _urlEndpoint);
98 }
99 NSUserDefaults *userDefaults = [NSUserDefaults standardUserDefaults];
100 [userDefaults setObject:endpoint forKey:kSPErrorTrackerUrl];
101 [userDefaults setObject:urlSuffix forKey:kSPErrorTrackerProtocol];
102 [userDefaults setObject:urlPrefix forKey:kSPErrorTrackerMethod];
103
104 _builderFinished = YES;
105}

160- (NSArray<SPRequestResult *> *)sendRequests:(NSArray<SPRequest *> *)requests {
161 NSMutableArray<SPRequestResult *> *results = [NSMutableArray new];
162
163 for (SPRequest *request in requests) {
164 NSMutableURLRequest *urlRequest = _httpMethod == SPHttpMethodGet
165 ? [self buildGetRequest:request]
166 : [self buildPostRequest:request];
167
168 [_dataOperationQueue addOperationWithBlock:^{
169 //source: https://forums.developer.apple.com/thread/11519
170 __block NSHTTPURLResponse *httpResponse = nil;
171 __block NSError *connectionError = nil;
172 dispatch_semaphore_t sem;
173
174 sem = dispatch_semaphore_create(0);
175
176 [[[NSURLSession sharedSession] dataTaskWithRequest:urlRequest
177 completionHandler:^(NSData *data, NSURLResponse *urlResponse, NSError *error) {
178
179 connectionError = error;
180 httpResponse = (NSHTTPURLResponse*)urlResponse;
181 dispatch_semaphore_signal(sem);
182 }] resume];
183
184 dispatch_semaphore_wait(sem, DISPATCH_TIME_FOREVER);
185
186 SPRequestResult *result = [[SPRequestResult alloc] initWithStatusCode:[httpResponse statusCode] oversize:request.oversize storeIds:request.emitterEventIds];
187 if (![result isSuccessful]) {
188 SPLogError(@"Connection error: %@", connectionError);
189 }
190
191 @synchronized (results) {
192 [results addObject:result];
193 }
194 }];
195 }
196 [_dataOperationQueue waitUntilAllOperationsAreFinished];
197 return results;
198}

202- (NSMutableURLRequest *)buildPostRequest:(SPRequest *)request {
203 NSData *requestData = [NSJSONSerialization dataWithJSONObject:[request.payload getAsDictionary] options:0 error:nil];
204 NSMutableURLRequest *urlRequest = [NSMutableURLRequest requestWithURL:[NSURL URLWithString:_urlEndpoint.absoluteString]];
205 [urlRequest setValue:[NSString stringWithFormat:@"%@", @(requestData.length).stringValue] forHTTPHeaderField:@"Content-Length"];
206 [urlRequest setValue:kSPAcceptContentHeader forHTTPHeaderField:@"Accept"];
207 [urlRequest setValue:kSPContentTypeHeader forHTTPHeaderField:@"Content-Type"];
208 if (_serverAnonymisation) {
209 [urlRequest setValue:@"*" forHTTPHeaderField:@"SP-Anonymous"];
210 }
211 [self applyValuesAndHeaderFields:_requestHeaders toRequest:urlRequest];
212 [urlRequest setHTTPMethod:@"POST"];
213 [urlRequest setHTTPBody:requestData];
214 return urlRequest;
215}

217- (NSMutableURLRequest *)buildGetRequest:(SPRequest *)request {
218 NSDictionary<NSString *, NSObject *> *payload = [request.payload getAsDictionary];
219 NSString *url = [NSString stringWithFormat:@"%@?%@", _urlEndpoint.absoluteString, [SPUtilities urlEncodeDictionary:payload]];
220 NSMutableURLRequest *urlRequest = [NSMutableURLRequest requestWithURL:[NSURL URLWithString:url]];
221 [urlRequest setValue:kSPAcceptContentHeader forHTTPHeaderField:@"Accept"];
222 if (_serverAnonymisation) {
223 [urlRequest setValue:@"*" forHTTPHeaderField:@"SP-Anonymous"];
224 }
225 [self applyValuesAndHeaderFields:_requestHeaders toRequest:urlRequest];
226 [urlRequest setHTTPMethod:@"GET"];
227 return urlRequest;
228}

230- (void)applyValuesAndHeaderFields:(NSDictionary<NSString *, NSString *> *)requestHeaders toRequest:(NSMutableURLRequest *)request {
231 [requestHeaders enumerateKeysAndObjectsUsingBlock:^(NSString * _Nonnull key, NSString * _Nonnull obj, BOOL * _Nonnull stop) {
232 [request setValue:obj forHTTPHeaderField:key];
233 }];
