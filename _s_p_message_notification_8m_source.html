89- (NSDictionary<NSString *, NSObject *> *)payload {
90 NSMutableDictionary *payload = [NSMutableDictionary dictionary];

92 if (_attachments.count > 0) {

95 [dictAttachments addObject:attachment.data];

99 [payload setValue:_body forKey:kSPMessageNotificationParamBody];
100 if (_bodyLocArgs.count > 0) {
101 [payload setValue:_bodyLocArgs forKey:kSPMessageNotificationParamBodyLocArgs];

103 [payload setValue:_bodyLocKey forKey:kSPMessageNotificationParamBodyLocKey];
104 [payload setValue:_category forKey:kSPMessageNotificationParamCategory];

110 [payload setValue:_icon forKey:kSPMessageNotificationParamIcon];
111 [payload setValue:_notificationCount forKey:kSPMessageNotificationParamNotificationCount];
112 [payload setValue:_notificationTimestamp forKey:kSPMessageNotificationParamNotificationTimestamp];

115 [payload setValue:_tag forKey:kSPMessageNotificationParamTag];

128+ (SPMessageNotification *)messageNotificationWithUserInfo:(NSDictionary *)userInfo defaultTitle:(NSString *)defaultTitle defaultBody:(NSString *)defaultBody {
129 NSDictionary *aps = [userInfo sp_dictionaryForKey:@"aps" defaultValue:nil];
130 if (!aps) {
131 return nil;
132 }
133 NSDictionary *alert = [aps sp_dictionaryForKey:@"alert" defaultValue:nil];
134 if (!alert) {
135 return nil;
136 }
137 // alert fields
138 NSString *title = [alert sp_stringForKey:@"title" defaultValue:defaultTitle];
139 NSString *body = [alert sp_stringForKey:@"body" defaultValue:defaultBody];
140 if (!title || !body) {
141 return nil;
142 }
143 NSString *subtitle = [alert sp_stringForKey:@"subtitle" defaultValue:nil];
144 NSString *launchImage = [alert sp_stringForKey:@"launch-image" defaultValue:nil];
145 NSString *titleLocKey = [alert sp_stringForKey:@"title-loc-key" defaultValue:nil];
146 NSArray *titleLocArgs = [alert sp_arrayForKey:@"title-loc-args" itemClass:NSString.class defaultValue:nil];
147 NSString *locKey = [alert sp_stringForKey:@"loc-key" defaultValue:nil];
148 NSArray *locArgs = [alert sp_arrayForKey:@"loc-args" itemClass:NSString.class defaultValue:nil];
149 // aps fields
150 NSNumber *badge = [aps sp_numberForKey:@"badge" defaultValue:nil];
151 NSString *sound = [aps sp_stringForKey:@"sound" defaultValue:nil];
152 NSNumber *contentAvailable = [aps sp_numberForKey:@"content-available" defaultValue:nil];
153 NSString *category = [aps sp_stringForKey:@"category" defaultValue:nil];
154 NSString *threadId = [aps sp_stringForKey:@"thread-id" defaultValue:nil];
155 SPMessageNotification *event = [[SPMessageNotification alloc] initWithTitle:title body:body trigger:SPMessageNotificationTriggerPush];
156 event.subtitle = subtitle;
157 event.icon = launchImage;
158 event.titleLocKey = titleLocKey;
159 event.titleLocArgs = titleLocArgs;
160 event.bodyLocKey = locKey;
161 event.bodyLocArgs = locArgs;
162 event.notificationCount = badge;
163 event.sound = sound;
164 event.contentAvailable = contentAvailable;
165 event.category = category;
166 event.threadIdentifier = threadId;
167 return event;
168}
