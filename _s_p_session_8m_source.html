208 - (BOOL) writeSessionToFile {
209  NSArray *paths = NSSearchPathForDirectoriesInDomains(NSDocumentDirectory, NSUserDomainMask, YES);
210  BOOL result = NO;
211  if ([paths count] > 0) {
212  NSString * savePath = [[paths lastObject] stringByAppendingPathComponent:kSessionSavePath];
213  NSMutableDictionary * sessionDict = [NSMutableDictionary dictionaryWithDictionary:[_sessionDict copy]];
214  [sessionDict removeObjectForKey:kSPSessionPreviousId];
215  [sessionDict removeObjectForKey:kSPSessionStorage];
216  result = [sessionDict writeToFile:savePath atomically:YES];
217  }

231 - (void) checkSession:(NSTimer *)timer {
232  __weak __typeof__(self) weakSelf = self;
233 
234  dispatch_async(_sessionQueue, ^{
235  __typeof__(self) strongSelf = weakSelf;
236  if (strongSelf == nil) return;
237 
238  NSNumber *checkTime = [SPUtilities getTimestamp];
239  NSInteger range = 0;
240 
241  if (strongSelf->_inBackground) {
242  range = strongSelf->_backgroundTimeout;
243  } else {
244  range = strongSelf->_foregroundTimeout;
245  }
246 
247  if (![strongSelf isTimeInRangeWithStartTime:strongSelf.accessedLast.longLongValue andCheckTime:checkTime.longLongValue andRange:range]) {
248  [strongSelf updateSession];
249  [strongSelf updateAccessedLast];
250  [strongSelf updateSessionDict];
251  [strongSelf writeSessionToFile];
252  }
253  });
254 }
