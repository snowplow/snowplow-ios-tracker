52/*
53 The IDFA can be retrieved using selectors rather than proper instance methods because
54 the compiler would complain about the missing AdSupport framework.
55 As stated in the header file, this only works if you have the AdSupport library in your project.
56 If you have it and you want to use IDFA, add the compiler flag <code>SNOWPLOW_IDFA_ENABLED</code> to your build settings.
57 If you haven't AdSupport framework in your project or SNOWPLOW_IDFA_ENABLED it's not set, it just compiles returning a nil advertisingIdentifier.
58
59 Note that `advertisingIdentifier` returns a sequence of 0s when used in the simulator.
60 Use a real device if you want a proper IDFA.
61 */
62- (NSString *) appleIdfa {
63#if SNOWPLOW_TARGET_IOS || SNOWPLOW_TARGET_TV
64#ifdef SNOWPLOW_IDFA_ENABLED
65 NSString *errorMsg = @"ASIdentifierManager not found. Please, add the AdSupport.framework if you want to use it.";
66 Class identifierManagerClass = NSClassFromString(@"ASIdentifierManager");
67 if (!identifierManagerClass) {
68 SPLogError(errorMsg);
69 return nil;
70 }
71
72 SEL sharedManagerSelector = NSSelectorFromString(@"sharedManager");
73 if (![identifierManagerClass respondsToSelector:sharedManagerSelector]) {
74 SPLogError(errorMsg);
75 return nil;
76 }
77
78 id identifierManager = ((id (*)(id, SEL))[identifierManagerClass methodForSelector:sharedManagerSelector])(identifierManagerClass, sharedManagerSelector);
79
80 if (@available(iOS 14.0, *)) {
81 errorMsg = @"ATTrackingManager not found. Please, add the AppTrackingTransparency.framework if you want to use it.";
82 Class trackingManagerClass = NSClassFromString(@"ATTrackingManager");
83 if (!trackingManagerClass) {
84 SPLogError(errorMsg);
85 return nil;
86 }
87
88 SEL trackingStatusSelector = NSSelectorFromString(@"trackingAuthorizationStatus");
89 if (![trackingManagerClass respondsToSelector:trackingStatusSelector]) {
90 SPLogError(errorMsg);
91 return nil;
92 }
93
94 //notDetermined = 0, restricted = 1, denied = 2, authorized = 3
95 NSInteger authorizationStatus = ((NSInteger (*)(id, SEL))[trackingManagerClass methodForSelector:trackingStatusSelector])(trackingManagerClass, trackingStatusSelector);
96
97 if (authorizationStatus != 3) {
98 SPLogDebug(@"The user didn't let tracking of IDFA. Authorization status is: %d", authorizationStatus);
99 return nil;
100 }
101 } else {
102 SEL isAdvertisingTrackingEnabledSelector = NSSelectorFromString(@"isAdvertisingTrackingEnabled");
103 if (![identifierManager respondsToSelector:isAdvertisingTrackingEnabledSelector]) {
104 SPLogError(errorMsg);
105 return nil;
106 }
107
108 BOOL isAdvertisingTrackingEnabled = ((BOOL (*)(id, SEL))[identifierManager methodForSelector:isAdvertisingTrackingEnabledSelector])(identifierManager, isAdvertisingTrackingEnabledSelector);
109 if (!isAdvertisingTrackingEnabled) {
110 SPLogError(@"The user didn't let tracking of IDFA.");
111 return nil;
112 }
113 }
114
115 SEL advertisingIdentifierSelector = NSSelectorFromString(@"advertisingIdentifier");
116 if (![identifierManager respondsToSelector:advertisingIdentifierSelector]) {
117 SPLogError(@"ASIdentifierManager doesn't respond to selector `advertisingIdentifier`.");
118 return nil;
119 }
120
121 NSUUID *uuid = ((NSUUID* (*)(id, SEL))[identifierManager methodForSelector:advertisingIdentifierSelector])(identifierManager, advertisingIdentifierSelector);
122 return [uuid UUIDString];
123#endif
124#endif
125 return nil;
126}

142- (NSString *) deviceModel {
143 NSString *simulatorModel = [NSProcessInfo.processInfo.environment objectForKey: @"SIMULATOR_MODEL_IDENTIFIER"];
144 if (simulatorModel) return simulatorModel;
145
146 size_t size;
147 sysctlbyname("hw.machine", NULL, &size, NULL, 0);
148 char *machine = malloc(size);
149 sysctlbyname("hw.machine", machine, &size, NULL, 0);
150 NSString *platform = [NSString stringWithUTF8String:machine];
151 free(machine);
152 return platform;
153}

158#elif SNOWPLOW_TARGET_WATCHOS
159 return [[WKInterfaceDevice currentDevice] systemVersion];

161 SInt32 osxMajorVersion;
162 SInt32 osxMinorVersion;
163 SInt32 osxPatchFixVersion;
164 NSProcessInfo *info = [NSProcessInfo processInfo];
165 NSOperatingSystemVersion systemVersion = [info operatingSystemVersion];
166 osxMajorVersion = (int)systemVersion.majorVersion;
167 osxMinorVersion = (int)systemVersion.minorVersion;
168 osxPatchFixVersion = (int)systemVersion.patchVersion;
169 NSString *versionString = [NSString stringWithFormat:@"%d.%d.%d", osxMajorVersion,
170 osxMinorVersion, osxPatchFixVersion];
171 return versionString;
172#endif

187- (NSString *) carrierName {
188#if SNOWPLOW_TARGET_IOS
189 CTTelephonyNetworkInfo *networkInfo = [CTTelephonyNetworkInfo new];
190 CTCarrier *carrier;
191 if (@available(iOS 12.1, *)) {
192 // `serviceSubscribersCellularProviders` has a bug in the iOS 12.0 so we use it from iOS 12.1
193 NSString *carrierKey = [self carrierKey];
194 if (!carrierKey) {
195 return nil;
196 }
197 NSDictionary<NSString *,CTCarrier *> *services = [networkInfo serviceSubscriberCellularProviders];
198 carrier = services[carrierKey];
199 } else {
200 carrier = [networkInfo subscriberCellularProvider];
201 }
202 return [carrier carrierName];
203#endif
204 return nil;
205}

225- (NSString *) carrierKey {
226#if SNOWPLOW_TARGET_IOS
227 if (@available(iOS 12.1, *)) {
228 CTTelephonyNetworkInfo *networkInfo = [CTTelephonyNetworkInfo new];
229 // `serviceSubscribersCellularProviders` has a bug in the iOS 12.0 so we use it from iOS 12.1
230 NSDictionary<NSString *,CTCarrier *> *services = [networkInfo serviceSubscriberCellularProviders];
231 NSArray<NSString *> *carrierKeys = services.allKeys;
232 // From iOS 12, iPhones with eSIMs can return multiple carrier providers.
233 // We can't prefer anyone of them so we track the first reported.
234 return carrierKeys.firstObject;
235 }
236#endif
237 return nil;
238}
