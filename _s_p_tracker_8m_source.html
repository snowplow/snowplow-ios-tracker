79void uncaughtExceptionHandler(NSException *exception) {
80 NSArray* symbols = [exception callStackSymbols];
81 NSString * stacktrace = [NSString stringWithFormat:@"Stacktrace:\n%@", symbols];
82 NSString * message = [exception reason];
83 dispatch_sync(dispatch_get_global_queue(DISPATCH_QUEUE_PRIORITY_DEFAULT, 0), ^{
84 if (message == nil || [message length] == 0) {
85 return;
86 }
87
88 // Construct userInfo
89 NSMutableDictionary<NSString *, NSObject *> *userInfo = [NSMutableDictionary new];
90 userInfo[@"message"] = message;
91 userInfo[@"stacktrace"] = stacktrace;
92
93 // Send notification to tracker
94 [[NSNotificationCenter defaultCenter]
95 postNotificationName:@"SPCrashReporting"
96 object:nil
97 userInfo:userInfo];
98
99 [NSThread sleepForTimeInterval:2.0f];
100 });
101}

236- (void)checkInstall {
237 if (_installEvent) {
238 SPInstallTracker * installTracker = [[SPInstallTracker alloc] init];
239 NSDate *previousTimestamp = installTracker.previousInstallTimestamp;
240 [installTracker clearPreviousInstallTimestamp];
241 if (!installTracker.isNewInstall && previousTimestamp == nil) {
242 return;
243 }
244 SPSelfDescribingJson *installEvent = [[SPSelfDescribingJson alloc] initWithSchema:kSPApplicationInstallSchema andData:@{}];
245 SPSelfDescribing *event = [[SPSelfDescribing alloc] initWithEventData:installEvent];
246 event.trueTimestamp = previousTimestamp; // it can be nil
247 [self track:event];
248 }

251- (void) setTrackerData {
252 NSString *trackerVersion = kSPVersion;
253 if ([_trackerVersionSuffix length]) {
254 NSMutableCharacterSet *allowedCharSet = [NSMutableCharacterSet alphanumericCharacterSet];
255 [allowedCharSet formUnionWithCharacterSet:[NSCharacterSet characterSetWithCharactersInString:@".-"]];
256 NSString *suffix = [[_trackerVersionSuffix componentsSeparatedByCharactersInSet:[allowedCharSet invertedSet]] componentsJoinedByString:@""];
257 if ([suffix length]) {
258 trackerVersion = [NSString stringWithFormat:@"%@ %@", trackerVersion, suffix];
259 }
260 }
261 _trackerData = [NSMutableDictionary dictionaryWithObjectsAndKeys:
262 trackerVersion, kSPTrackerVersion,
263 _trackerNamespace, kSPNamespace,
264 _appId != nil ? _appId : [NSNull null], kSPAppId, nil];
265}

534- (void)processEvent:(SPEvent *)event {
535 SPTrackerState *stateSnapshot;
536 @synchronized (self) {
537 stateSnapshot = [self.stateManager trackerStateForProcessedEvent:event];
538 }
539 SPTrackerEvent *trackerEvent = [[SPTrackerEvent alloc] initWithEvent:event state:stateSnapshot];
540 [self transformEvent:trackerEvent];
541 SPPayload *payload = [self payloadWithEvent:trackerEvent];
542 [_emitter addPayloadToBuffer:payload];
543}

545- (void)transformEvent:(SPTrackerEvent *)event {
546 // Application_install event needs the timestamp to the real installation event.
547 if ([event.schema isEqualToString:kSPApplicationInstallSchema] && event.trueTimestamp) {
548 event.timestamp = event.trueTimestamp.timeIntervalSince1970 * 1000;
549 event.trueTimestamp = nil;
550 }
551 // Payload can be optionally updated with values based on internal state
552 [self.stateManager addPayloadValuesToEvent:event];
553}

555- (SPPayload *)payloadWithEvent:(SPTrackerEvent *)event {
556 SPPayload *payload = [SPPayload new];
557 payload.allowDiagnostic = !event.isService;
558
559 [self addBasicPropertiesToPayload:payload event:event];
560 if (event.isPrimitive) {
561 [self addPrimitivePropertiesToPayload:payload event:event];
562 } else {
563 [self addSelfDescribingPropertiesToPayload:payload event:event];
564 }
565 NSMutableArray<SPSelfDescribingJson *> *contexts = event.contexts;
566 [self addBasicContextsToContexts:contexts event:event];
567 [self addGlobalContextsToContexts:contexts event:event];
568 [self addStateMachineEntitiesToContexts:contexts event:event];
569 [self wrapContexts:contexts toPayload:payload];
570 return payload;
571}

608/*
609 This is needed because the campaign-attribution-enrichment (in the pipeline) is able to parse
610 the `url` and `referrer` only if they are part of a PageView event.
611 The PageView event is an atomic event but the DeepLinkReceived is a SelfDescribing event.
612 For this reason we copy these two fields in the atomic fields in order to let the enrichment
613 to process correctly the fields even if the event is not a PageView and it's a SelfDescribing event.
614 This is a hack that should be removed once the atomic event table is dismissed and all the events
615 will be SelfDescribing.
616 */
617- (void)workaroundForCampaignAttributionEnrichment:(SPPayload *)payload event:(SPTrackerEvent *)event {
618 if ([event.schema isEqualToString:kSPDeepLinkReceivedSchema]) {
619 NSString *url = (NSString *)[event.payload objectForKey:kSPDeepLinkReceivedParamUrl];
620 NSString *referrer = (NSString *)[event.payload objectForKey:kSPDeepLinkReceivedParamReferrer];
621 [payload addValueToPayload:url forKey:kSPPageUrl];
622 [payload addValueToPayload:referrer forKey:kSPPageRefr];
623 }

630- (void)addBasicContextsToContexts:(NSMutableArray<SPSelfDescribingJson *> *)contexts eventId:(NSString *)eventId eventTimestamp:(long long)eventTimestamp isService:(BOOL)isService {
631 if (_subject) {
632 NSDictionary * platformDict = [[_subject getPlatformDict] getAsDictionary];
633 if (platformDict != nil) {
634 [contexts addObject:[[SPSelfDescribingJson alloc] initWithSchema:_platformContextSchema andData:platformDict]];
635 }
636 NSDictionary * geoLocationDict = [_subject getGeoLocationDict];
637 if (geoLocationDict != nil) {
638 [contexts addObject:[[SPSelfDescribingJson alloc] initWithSchema:kSPGeoContextSchema andData:geoLocationDict]];
639 }
640 }
641
642 if (_applicationContext) {
643 SPSelfDescribingJson * contextJson = [SPUtilities getApplicationContext];
644 if (contextJson != nil) {
645 [contexts addObject:contextJson];
646 }
647 }
648
649 if (isService) {
650 return;
651 }
652
653 // Add session
654 if (_session) {
655 NSDictionary *sessionDict = [_session getSessionDictWithEventId:eventId eventTimestamp:eventTimestamp];
656 if (sessionDict) {
657 [contexts addObject:[[SPSelfDescribingJson alloc] initWithSchema:kSPSessionContextSchema andData:sessionDict]];
658 } else {
659 SPLogTrack(nil, @"Unable to get session context for eventId: %@", eventId);
660 }
661 }
662
663 // Add GDPR context
664 if (self.gdpr) {
665 SPSelfDescribingJson *gdprContext = self.gdpr.context;
666 if (gdprContext) {
667 [contexts addObject:gdprContext];
668 }
669 }
670}

683- (void)wrapContexts:(NSArray<SPSelfDescribingJson *> *)contexts toPayload:(SPPayload *)payload {
684 if (contexts.count == 0) {
685 return;
686 }
687 NSMutableArray<NSDictionary *> *data = [NSMutableArray new];
688 for (SPSelfDescribingJson *context in contexts) {
689 [data addObject:[context getAsDictionary]];
690 }
691
692 SPSelfDescribingJson *finalContext = [[SPSelfDescribingJson alloc] initWithSchema:kSPContextSchema andData:data];
693 if (finalContext == nil) {
694 return;
695 }
696 [payload addDictionaryToPayload:[finalContext getAsDictionary]
697 base64Encoded:_base64Encoded
698 typeWhenEncoded:kSPContextEncoded
699 typeWhenNotEncoded:kSPContext];
700}
