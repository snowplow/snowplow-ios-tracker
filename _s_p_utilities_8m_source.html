154+ (NSDictionary *) replaceHyphenatedKeysWithCamelcase:(NSDictionary *)dict{
155 NSMutableDictionary * newDictionary = [[NSMutableDictionary alloc] init];
156 for (NSString * key in dict) {
157 if ([self string:key contains:@"-"]) {
158 if ([dict[key] isKindOfClass:[NSDictionary class]]) {
159 newDictionary[[self camelcaseParsedKey:key]] = [self replaceHyphenatedKeysWithCamelcase:dict[key]];
160 } else {
161 newDictionary[[self camelcaseParsedKey:key]] = dict[key];
162 }
163 } else {
164 if ([dict[key] isKindOfClass:[NSDictionary class]]) {
165 newDictionary[key] = [self replaceHyphenatedKeysWithCamelcase:dict[key]];
166 } else {
167 newDictionary[key] = dict[key];
168 }
169 }
170 }
171
172 return [[NSDictionary alloc] initWithDictionary:newDictionary copyItems:YES];
173}

184+ (NSString *) camelcaseParsedKey:(NSString *)key {
185 NSScanner * scanner = [[NSScanner alloc] initWithString:key];
186 NSMutableArray * words = [[NSMutableArray alloc] init];
187 NSString * scannedWord = [[NSString alloc] init];
188
189 while (![scanner isAtEnd]) {
190 [scanner scanUpToString:@"-" intoString:&scannedWord];
191 [words addObject:scannedWord];
192 SPLogVerbose(@"scanned word: %@", scannedWord);
193 [scanner scanString:@"-" intoString:nil];
194 }
195
196 SPLogVerbose(@"%@", words);
197 if ([words count] == 0) {
198 return @"";
199 } else if ([words count] == 1) {
200 return [[NSString alloc] initWithString:[words[0] lowercaseString]];
201 } else {
202 NSMutableString * camelcaseKey = [[NSMutableString alloc] initWithString:[words[0] lowercaseString]];
203 NSRange range;
204 range.length = words.count-1;
205 range.location = 1;
206 for (NSString * word in [words subarrayWithRange:range]) {
207 [camelcaseKey appendString:[word capitalizedString]];
208 }
209 return camelcaseKey;
210 }
211}
