88 /*
89  The IDFA can be retrieved using selectors rather than proper instance methods because
90  the compiler would complain about the missing AdSupport framework.
91  As stated in the header file, this only works if you have the AdSupport library in your project.
92  If you have it and you want to use IDFA, add the compiler flag <code>SNOWPLOW_IDFA_ENABLED</code> to your build settings.
93  If you haven't AdSupport framework in your project or SNOWPLOW_IDFA_ENABLED it's not set, it just compiles returning a nil advertisingIdentifier.
94 
95  Note that `advertisingIdentifier` returns a sequence of 0s when used in the simulator.
96  Use a real device if you want a proper IDFA.
97  */
98 + (NSString *) getAppleIdfa {
99 #if SNOWPLOW_TARGET_IOS || SNOWPLOW_TARGET_TV
100 #ifdef SNOWPLOW_IDFA_ENABLED
101  NSString *errorMsg = @"ASIdentifierManager not found. Please, add the AdSupport.framework if you want to use it.";
102  Class identifierManagerClass = NSClassFromString(@"ASIdentifierManager");
103  if (!identifierManagerClass) {
104  SPLogError(errorMsg);
105  return nil;
106  }
107 
108  SEL sharedManagerSelector = NSSelectorFromString(@"sharedManager");
109  if (![identifierManagerClass respondsToSelector:sharedManagerSelector]) {
110  SPLogError(errorMsg);
111  return nil;
112  }
113 
114  id identifierManager = ((id (*)(id, SEL))[identifierManagerClass methodForSelector:sharedManagerSelector])(identifierManagerClass, sharedManagerSelector);
115 
116  if (@available(iOS 14.0, *)) {
117  errorMsg = @"ATTrackingManager not found. Please, add the AppTrackingTransparency.framework if you want to use it.";
118  Class trackingManagerClass = NSClassFromString(@"ATTrackingManager");
119  if (!trackingManagerClass) {
120  SPLogError(errorMsg);
121  return nil;
122  }
123 
124  SEL trackingStatusSelector = NSSelectorFromString(@"trackingAuthorizationStatus");
125  if (![trackingManagerClass respondsToSelector:trackingStatusSelector]) {
126  SPLogError(errorMsg);
127  return nil;
128  }
129 
130  //notDetermined = 0, restricted = 1, denied = 2, authorized = 3
131  NSInteger authorizationStatus = ((NSInteger (*)(id, SEL))[trackingManagerClass methodForSelector:trackingStatusSelector])(trackingManagerClass, trackingStatusSelector);
132 
133  if (authorizationStatus != 3) {
134  SPLogDebug(@"The user didn't let tracking of IDFA. Authorization status is: %d", authorizationStatus);
135  return nil;
136  }
137  } else {
138  SEL isAdvertisingTrackingEnabledSelector = NSSelectorFromString(@"isAdvertisingTrackingEnabled");
139  if (![identifierManager respondsToSelector:isAdvertisingTrackingEnabledSelector]) {
140  SPLogError(errorMsg);
141  return nil;
142  }
143 
144  BOOL isAdvertisingTrackingEnabled = ((BOOL (*)(id, SEL))[identifierManager methodForSelector:isAdvertisingTrackingEnabledSelector])(identifierManager, isAdvertisingTrackingEnabledSelector);
145  if (!isAdvertisingTrackingEnabled) {
146  SPLogError(@"The user didn't let tracking of IDFA.");
147  return nil;
148  }
149  }
150 
151  SEL advertisingIdentifierSelector = NSSelectorFromString(@"advertisingIdentifier");
152  if (![identifierManager respondsToSelector:advertisingIdentifierSelector]) {
153  SPLogError(@"ASIdentifierManager doesn't respond to selector `advertisingIdentifier`.");
154  return nil;
155  }
156 
157  NSUUID *uuid = ((NSUUID* (*)(id, SEL))[identifierManager methodForSelector:advertisingIdentifierSelector])(identifierManager, advertisingIdentifierSelector);
158  return [uuid UUIDString];
159 #endif
160 #endif
161  return nil;
162 }

400  NSMutableDictionary * newDictionary = [[NSMutableDictionary alloc] init];

429 + (NSString *) camelcaseParsedKey:(NSString *)key {
430  NSScanner * scanner = [[NSScanner alloc] initWithString:key];
431  NSMutableArray * words = [[NSMutableArray alloc] init];
432  NSString * scannedWord = [[NSString alloc] init];
433 
434  while (![scanner isAtEnd]) {
435  [scanner scanUpToString:@"-" intoString:&scannedWord];
436  [words addObject:scannedWord];
437  SPLogVerbose(@"scanned word: %@", scannedWord);
438  [scanner scanString:@"-" intoString:nil];
439  }
440 
441  SPLogVerbose(@"%@", words);
442  if ([words count] == 0) {
443  return @"";
444  } else if ([words count] == 1) {
445  return [[NSString alloc] initWithString:[words[0] lowercaseString]];
446  } else {
447  NSMutableString * camelcaseKey = [[NSMutableString alloc] initWithString:[words[0] lowercaseString]];
448  NSRange range;
449  range.length = words.count-1;
450  range.location = 1;
451  for (NSString * word in [words subarrayWithRange:range]) {
452  [camelcaseKey appendString:[word capitalizedString]];
453  }
454  return camelcaseKey;
455  }
456 }
