92 - (void)log:(SPLogLevel)level tag:(NSString *)tag message:(NSString *)message {
93  if (level > self.logLevel) {
94  return;
95  }
96  if (self.delegate) {
97  switch (level) {
98  case SPLogLevelOff:
99  // do nothing.
100  break;
101  case SPLogLevelError:
102  [self.delegate error:tag message:message];
103  break;
104  case SPLogLevelDebug:
105  [self.delegate debug:tag message:message];
106  break;
107  case SPLogLevelVerbose:
108  [self.delegate verbose:tag message:message];
109  break;
110  }
111  return;
112  }
113  #if SNOWPLOW_TEST
114  // NSLog doesn't work on test target
115  NSString *output = [NSString stringWithFormat:@"[%@] %@: %@", @[@"Off", @"Error", @"Error", @"Debug", @"Verbose"][level], tag, message];
116  printf("%s", [output UTF8String]);
117  #elif DEBUG
118  // Log should be printed only during debugging
119  NSLog(@"[%@] %@: %@", @[@"Off", @"Error", @"Debug", @"Verbose"][level], tag, message);
120  #endif
121 }

123 - (void)trackErrorWithTag:(NSString *)tag message:(NSString *)message errorOrException:(id)errorOrException {
124  NSError *error;
125  NSException *exception;
126  if ([errorOrException isKindOfClass:NSError.class]) {
127  error = (NSError *)errorOrException;
128  } else if ([errorOrException isKindOfClass:NSException.class]) {
129  exception = (NSException *)errorOrException;
130  }
131 
132  // Construct userInfo
133  NSMutableDictionary<NSString *, NSObject *> *userInfo = [NSMutableDictionary new];
134  userInfo[@"tag"] = tag;
135  userInfo[@"message"] = message;
136  userInfo[@"error"] = error;
137  userInfo[@"exception"] = exception;
138 
139  // Send notification to tracker
140  [[NSNotificationCenter defaultCenter]
141  postNotificationName:@"SPTrackerDiagnostic"
142  object:self
143  userInfo:userInfo];
144 }
